### [Squarified](https://www.codechef.com/problems/SQUAR)

```
#include <bits/stdc++.h>
using namespace std;
#define INF (int)1e18
#define f first
#define s second

mt19937_64 RNG(chrono::steady_clock::now().time_since_epoch().count());

const int MAXN = 1e7 + 105;
int cache[MAXN];

void Solve() 
{
    int n; cin >> n;
    map <int, int> m;
    
    while (n--){
        int x; cin >> x;
        int sx = x;
        int p = 1;
        
        if (cache[sx] == 0) {
            for (int i = 2; i * i <= x; i++){
                int cnt = 0;
                while (x % i == 0){
                    x /= i;
                    cnt++;
                }
                
                if (cnt % 2 == 1) p *= i;
            }
            p *= x;
            cache[sx] = p;
        }
        
        if (cache[sx] != 1) m[cache[sx]]++;
    }
    
    int ans = 0;
    for (auto [x, y] : m) ans = max(ans, y);
    
    cout << ans << "\n";
}

int32_t main() 
{
    auto begin = std::chrono::high_resolution_clock::now();
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int t = 1;
    cin >> t;
    for(int i = 1; i <= t; i++) 
    {
        //cout << "Case #" << i << ": ";
        Solve();
    }
    auto end = std::chrono::high_resolution_clock::now();
    auto elapsed = std::chrono::duration_cast<std::chrono::nanoseconds>(end - begin);
    cerr << "Time measured: " << elapsed.count() * 1e-9 << " seconds.\n"; 
    return 0;
}
```
